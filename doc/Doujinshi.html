def initialize(id)
  @id           = id
  @client       = Net::HTTP.get_response(URI("https://nhentai.net/g/#{@id}/"))
  @media_id     = @client.body.match(%r{\/([0-9]+)\/cover.jpg})[1]
  @count_pages  = @client.body.match(/([0-9]+) pages/)[1].to_i
end
